CREATE TABLE AuditLog (
    LogID NUMBER,
    TransactionID NUMBER,
    LogDate DATE,
    Message VARCHAR2(200)
);

CREATE OR REPLACE TRIGGER LogTransaction
AFTER INSERT ON Transactions
FOR EACH ROW
BEGIN
    INSERT INTO AuditLog VALUES (
        :NEW.TransactionID,
        :NEW.TransactionID,
        SYSDATE,
        'Transaction Inserted'
    );
END;
